\begin{theorem}[Quadratic Formula]
    Let $ax^{2}+bx +c =0$ be a quadratic equation. Then the solutions to the quadratic equation are given by 
		\[
			x = \frac{-b \pm \sqrt{b^{2} - 4ac}}{2a}
		\]
\end{theorem} \vfill
